main {
    auto_trigger_frame = 0;
    auto_trigger_btn   = 0;
    time_since_last_dribble += get_rtime();
    set_rgb(TEAL_R, TEAL_G, TEAL_B);

    if (get_console() == PIO_XB1) { isXbox = 1; } else { isXbox = 0; }

    if (startup && isLoadingScreen) {
        dt_ms = get_rtime();
        load_phase_ms += dt_ms;
        link_phase_ms += dt_ms;

        starfield_step();

        cls_oled(OLED_BLACK);
        starfield_draw();
        print(1, OLED_HEIGHT - 10, OLED_FONT_SMALL, OLED_WHITE, version_txt[0]);
        profile_idx = current_profile - 1;

        if (profile_idx < 0) profile_idx = 0;

        if (profile_idx > (PROFILE_MAX - 1)) profile_idx = PROFILE_MAX - 1;

        profile_width = (get_string_length(profile_labels[profile_idx]) - 1) * OLED_FONT_SMALL_WIDTH;
        profile_x     = OLED_WIDTH - profile_width - 7;
        profile_y     = OLED_HEIGHT - 10;

        print(profile_x, profile_y, OLED_FONT_SMALL, OLED_WHITE, profile_labels[profile_idx]);

        if (load_phase == 0) {
            print(av_x, av_y, OLED_FONT_MEDIUM, OLED_WHITE, brand_txt[0]);

            if (load_phase_ms >= LOAD_PHASE0_MS) {
                load_phase = 1;
                load_phase_ms = 0;
                av_vx = LOAD_SWOOSH_VX_START; 
            }
        }
    else if (load_phase == 1) {
        if (av_vx < av_vx_cap) av_vx += av_ax;
        av_x += av_vx;

        if (av_x < OLED_WIDTH && (av_x + brand_w_px) > 0) {
            print(av_x, av_y, OLED_FONT_MEDIUM, OLED_WHITE, brand_txt[0]);
        } else {
            av_x = OLED_WIDTH; 
            av_vx = 0;        
        }
        if (av_x >= OLED_WIDTH && load_phase_ms >= LOAD_SWOOSH_MIN_MS) {
            load_phase = 2;
            load_phase_ms = 0;
        }
    }
    else if (load_phase == 2) {
        print(link_line1_x, link_line1_y, OLED_FONT_SMALL, OLED_WHITE, link_txt[0]);
        print(link_line2_x, link_line2_y, OLED_FONT_SMALL, OLED_WHITE, link_txt[1]);

        if (link_phase == 0) {
            if (link_velocity_x < link_velocity_cap_in) {
                link_velocity_x += link_accel_x;
            }

            link_line1_x += link_velocity_x;
            link_line2_x += link_velocity_x;

            if (link_line1_x >= link_line1_center_x) {
                link_line1_x = link_line1_center_x;
            }
            if (link_line2_x >= link_line2_center_x) {
                link_line2_x = link_line2_center_x;
            }

            if (link_line1_x == link_line1_center_x && link_line2_x == link_line2_center_x) {
                link_phase = 1;
                link_phase_ms = 0;
                link_velocity_x = 4;
            }
        }
        else if (link_phase == 1) {
            if (link_phase_ms >= link_pause_ms) {
                link_phase = 2;
                link_phase_ms = 0;
            }
        }
        else { 
            if (link_velocity_x < link_velocity_cap_out) {
                link_velocity_x += link_accel_x;
            }

            link_line1_x += link_velocity_x;
            link_line2_x += link_velocity_x;

            if (link_line1_x > OLED_WIDTH) {
                load_phase = 3;
                load_phase_ms = 0;
            }
        }
    }

    if (load_phase == 3) {
        cls_oled(OLED_BLACK);
        isLoadingScreen = 0;
        startup = 0;
        saver_on = 1;
        saver_timer = 80;
        saver_init_static();
    }
} else {
    if (event_release(PS4_SQUARE)) {
        if (auto_quickstop_triggered) {
            combo_stop(initial_rx_flick);
            combo_run(complete_flick_up_rs_smooth);
            auto_quickstop_triggered = 0;
        }
    }
    if (event_press(PS4_CIRCLE) && menu_enabled) {
        menu_enabled = 0;
        combo_run(vibration_off);
        set_rgb(0,255,0);
        saver_on = 1;
        saver_init_static();

        set_pvar(PVAR_FLICK_ACTIVE, flick_up_shooting_active);
        set_pvar(PVAR_TEMPO_MS, delay_duration);
        set_pvar(PVAR_STABILITY, stabilization_variation);
        set_pvar(PVAR_DRIBBLE, flick_sequences_enabled);
        set_pvar(PVAR_REVERSE_MODE, reverse_flick_mode);
        set_pvar(PVAR_RS_SHOOT, rs_shooting_active);
        set_pvar(PVAR_REMAP_SQ, remap_square_active);
        set_pvar(PVAR_QS_MODE, quickstop_mode);
        set_pvar(PVAR_DUNK_METER, meter_dunk_active);
        set_pvar(PVAR_DASSIST, dAssist);
        last_menu_page = page;
        set_pvar(PVAR_LAST_PAGE, last_menu_page);
        set_pvar(PVAR_FADE_MS, fade_delay_ms);
        set_pvar(PVAR_FADE_TOGGLE, fade_toggle);
        set_pvar(PVAR_AUTO_REMAP, auto_remap_mode);
        set_pvar(PVAR_AUTO_SHOT_ACTIVE, Auto_shot_active);
        set_pvar(PVAR_AUTO_SHOT_DUR, Auto_shot_duration);
        set_pvar(PVAR_AUTO_TEMPO, Delay_duration_Insta);
    }
    else if (!menu_enabled && get_val(PS4_L2) && event_press(PS4_OPTIONS)) {
        menu_enabled = 1;
        page = last_menu_page;
        combo_run(vibration_on);
        set_rgb(0,0,255);
        saver_on = 0;
    }

    if (menu_enabled) {
        dt_menu = get_rtime();
        repeat_update(dt_menu);
        starfield_step();
        cls_oled(OLED_BLACK);
        starfield_draw();
        print(
            center_x(get_string_length(menu_header[page - 1]), OLED_FONT_SMALL_WIDTH),
            20, OLED_FONT_SMALL, OLED_WHITE,
            menu_header[page - 1]
        );

        if (page == 16) {
            print(
                center_x(get_string_length(reverse_modes[reverse_flick_mode]), OLED_FONT_MEDIUM_WIDTH),
                30, OLED_FONT_MEDIUM, OLED_WHITE,
                reverse_modes[reverse_flick_mode]
            );
        }

        if (page == 1 
            || page == 3  
            || page == 5   
            || page == 7  
            || page == 20 
            || page == 21  
        ) {
            print(
                center_x(get_string_length(onOff[menuSelection]), OLED_FONT_MEDIUM_WIDTH),
                30, OLED_FONT_MEDIUM, OLED_WHITE,
                onOff[menuSelection]
            );
        }
        if (page == 1) { pageChg_s = 0; } else if (page == numpgs) { pageChg_s = 2; } else { pageChg_s = 1; }
        if (page > numpgs) { page = numpgs; }
        if (page < 1) { page = 1; }
        if (get_val(PS4_L2)) { set_val(PS4_UP,0); set_val(PS4_DOWN,0); set_val(PS4_LEFT,0); set_val(PS4_RIGHT,0); }
        if (get_val(PS4_L2) && event_press(PS4_LEFT)) { page--; combo_run(vibration_valchg); }
        if (get_val(PS4_L2) && event_press(PS4_RIGHT)) { page++; combo_run(vibration_valchg); }
        if (get_val(PS4_L2) && (event_press(PS4_LEFT) || event_press(PS4_RIGHT))) {
            if (page < 1) page = 1;
            if (page > numpgs) page = numpgs;
            last_menu_page = page;
            set_pvar(PVAR_LAST_PAGE, last_menu_page);
        }
            // Page 1: Auto Green
            if (page == 1) {
                menuSelection = flick_up_shooting_active; toggleType = 0;
                if (get_val(PS4_L2) && event_press(PS4_UP)) { flick_up_shooting_active = 1; flashtoggle = 1; combo_run(vibration_valchg); combo_run(flashtog); }
                if (get_val(PS4_L2) && event_press(PS4_DOWN)) { flick_up_shooting_active = 0; flashtoggle = 0; combo_run(vibration_valchg); combo_run(flashtog); }
                set_pvar(PVAR_FLICK_ACTIVE, flick_up_shooting_active);
            }
            // Page 2: Tempo Speed (ms)
            if (page == 2) {
                menuSelection = 2; toggleType = 0;
                NumberToString(delay_duration, FindDigits(delay_duration));
                print(70, 30, OLED_FONT_MEDIUM, OLED_WHITE, ms[0]);
                draw_x_graph(delay_duration, 0, 300, 45);
                if (get_val(PS4_L2)) {
                    if (event_press(PS4_UP)   || repeat_up_due())   { delay_duration++; }
                    if (event_press(PS4_DOWN) || repeat_down_due()) { delay_duration--; }
                    if (event_press(PS4_UP) || event_press(PS4_DOWN) || repeat_up_due() || repeat_down_due()) {
                        combo_run(vibration_valchg); combo_run(flashedit);
                    }
                }
                if (delay_duration > 300) { delay_duration = 300; }
                if (delay_duration < 0) { delay_duration = 0; }
                set_pvar(PVAR_TEMPO_MS, delay_duration);
            }
            // Page 3: Fade Toggle
            if (page == 3) {
                menuSelection = fade_toggle; toggleType = 0;

                if (get_val(PS4_L2) && event_press(PS4_UP))   { fade_toggle = 1; flashtoggle = 1; combo_run(vibration_valchg); combo_run(flashtog); }
                if (get_val(PS4_L2) && event_press(PS4_DOWN)) { fade_toggle = 0; flashtoggle = 0; combo_run(vibration_valchg); combo_run(flashtog); }

                set_pvar(PVAR_FADE_TOGGLE, fade_toggle);
            }
            // Page 4: Fade (ms)
            if (page == 4) {
                menuSelection = 2; toggleType = 0;
                NumberToString(fade_delay_ms, FindDigits(fade_delay_ms));
                print(70, 30, OLED_FONT_MEDIUM, OLED_WHITE, ms[0]);
                draw_x_graph(fade_delay_ms, 0, 300, 45);

                if (get_val(PS4_L2)) {
                    if (event_press(PS4_UP) || repeat_up_due()) { fade_delay_ms++; }
                    if (event_press(PS4_DOWN) || repeat_down_due()) { fade_delay_ms--; }
                    if (event_press(PS4_UP) || event_press(PS4_DOWN) || repeat_up_due() || repeat_down_due()) {
                        combo_run(vibration_valchg); combo_run(flashedit);
                    }
                }
                if (fade_delay_ms > 300) { fade_delay_ms = 300; }
                if (fade_delay_ms <   0) { fade_delay_ms = 0; }
                set_pvar(PVAR_FADE_MS, fade_delay_ms);
            }
            // Page 5: No-Dip Shot (ON/OFF)
            if (page == 5) {
                menuSelection = NODIP_SHOT_ENABLED; toggleType = 0;
                if (get_val(PS4_L2) && event_press(PS4_UP)) { NODIP_SHOT_ENABLED = 1; flashtoggle = 1; combo_run(vibration_valchg); combo_run(flashtog); }
                if (get_val(PS4_L2) && event_press(PS4_DOWN)) { NODIP_SHOT_ENABLED = 0; flashtoggle = 0; combo_run(vibration_valchg); combo_run(flashtog); }
                set_pvar(PVAR_NODIP_ENABLED, NODIP_SHOT_ENABLED);
            }
            // Page 6: No-Dip Tempo 
            if (page == 6) {
                menuSelection = 2; toggleType = 0;
                NumberToString(NODIP_SHOT_DISPLAY_VALUE, FindDigits(NODIP_SHOT_DISPLAY_VALUE));
                print(70, 30, OLED_FONT_MEDIUM, OLED_WHITE, ms[0]);
                draw_x_graph(NODIP_SHOT_DISPLAY_VALUE, 1, 30, 45);

                if (get_val(PS4_L2)) {
                    if (event_press(PS4_UP) || repeat_up_due()) { NODIP_SHOT_DISPLAY_VALUE++; }
                    if (event_press(PS4_DOWN) || repeat_down_due()) { NODIP_SHOT_DISPLAY_VALUE--; }
                    if (event_press(PS4_UP) || event_press(PS4_DOWN) || repeat_up_due() || repeat_down_due()) {
                        combo_run(vibration_valchg); combo_run(flashedit);
                    }
                }
                if (NODIP_SHOT_DISPLAY_VALUE < 1) NODIP_SHOT_DISPLAY_VALUE = 1;
                if (NODIP_SHOT_DISPLAY_VALUE > 30) NODIP_SHOT_DISPLAY_VALUE = 30;

                update_nodip_window();
                set_pvar(PVAR_NODIP_VALUE, NODIP_SHOT_DISPLAY_VALUE);
            }
            // Page 7: Auto Shot (ON/OFF)
            if (page == 7) {
                menuSelection = Auto_shot_active; toggleType = 0;
                if (get_val(PS4_L2) && event_press(PS4_UP)) { Auto_shot_active = 1; flashtoggle = 1; combo_run(vibration_valchg); combo_run(flashtog); }
                if (get_val(PS4_L2) && event_press(PS4_DOWN)) { Auto_shot_active = 0; flashtoggle = 0; combo_run(vibration_valchg); combo_run(flashtog); }
                set_pvar(PVAR_AUTO_SHOT_ACTIVE, Auto_shot_active);
            }
            // Page 8: Auto Timing (ms)
            if (page == 8) {
                menuSelection = 2; toggleType = 0;

                    set_val(PS4_R2, 100);
                } else if (event_release(PS4_L1) || event_release(PS4_R2)) {
                    set_val(PS4_RY, 0);
                    set_val(PS4_R2, 0);
                }
            }
            if (dAssist) {
                if (get_val(PS4_L2) > 20 && get_val(PS4_R1) > 20) {
                    set_val(PS4_R1, 0);
                    if (abs(get_val(PS4_LX)) > 20 || abs(get_val(PS4_LY)) > 20) {
                        if (!combo_running(rs_dAssist)) {
                            combo_run(rs_dAssist);
                        }
                        flickedD = 1;
                    }
                } else {
                    if (flickedD) {
                        combo_stop(rs_dAssist);
                        flickedD = 0;
                    }
                    set_val(PS4_RX, get_val(PS4_RX));
                    set_val(PS4_RY, get_val(PS4_RY));
                }
            } else {
                set_val(PS4_RX, get_val(PS4_RX));
                set_val(PS4_RY, get_val(PS4_RY));
            }
        } 
    } 
}

function repeat_update(dt) {
    if (get_val(PS4_L2) && get_val(PS4_UP)) {
        if (!rpt_up_held) { rpt_up_held = 1; rpt_up_bucket = 0; rpt_up_total = 0; }
        rpt_up_bucket += dt;
        rpt_up_total  += dt;
    } else {
        rpt_up_held = 0; rpt_up_bucket = 0; rpt_up_total = 0;
    }
    if (get_val(PS4_L2) && get_val(PS4_DOWN)) {
        if (!rpt_dn_held) { rpt_dn_held = 1; rpt_dn_bucket = 0; rpt_dn_total = 0; }
        rpt_dn_bucket += dt;
        rpt_dn_total  += dt;
    } else {
        rpt_dn_held = 0; rpt_dn_bucket = 0; rpt_dn_total = 0;
    }
}

function repeat_up_due() {
    if (!rpt_up_held) return 0;
    if (rpt_up_total < REPEAT_INITIAL_MS) return 0;

    if (rpt_up_total >= REPEAT_ACCEL_AT) { rpt_rate_up = REPEAT_RATE_FAST; }
    else { rpt_rate_up = REPEAT_RATE_SLOW; }

    if (rpt_up_bucket >= rpt_rate_up) {
        rpt_up_bucket -= rpt_rate_up; 
        return 1;
    }
    return 0;
}

function repeat_down_due() {
    if (!rpt_dn_held) return 0;
    if (rpt_dn_total < REPEAT_INITIAL_MS) return 0;

    if (rpt_dn_total >= REPEAT_ACCEL_AT) { rpt_rate_dn = REPEAT_RATE_FAST; }
    else { rpt_rate_dn = REPEAT_RATE_SLOW; }

    if (rpt_dn_bucket >= rpt_rate_dn) {
        rpt_dn_bucket -= rpt_rate_dn;
        return 1;
    }
    return 0;
}

function starfield_step() {
    for (star_index = 0; star_index < STAR_COUNT; star_index++) {
        star_x[star_index] += star_vx[star_index];

            star_x[star_index]  = 0;
            star_y[star_index]  = random(0, OLED_HEIGHT - 1);
            star_vx[star_index] = 1 + random(0, 2);
        }
    }
}

int bufferIndex, digitIndex, charIndex;
function NumberToString(f_val, f_digits) {
    bufferIndex = 1;
    digitIndex = 10000;
    if (f_val < 0) { putc_oled(bufferIndex, 45); bufferIndex += 1; f_val = abs(f_val); }
    for (charIndex = 5; charIndex >= 1; charIndex--) {
        if (f_digits >= charIndex) {
            putc_oled(bufferIndex, (f_val / digitIndex) + 48);
            f_val %= digitIndex;
            bufferIndex++;
            if (charIndex == 4) { putc_oled(bufferIndex, 44); bufferIndex++; }
        }
        digitIndex /= 10;
    }
    puts_oled(center_x(bufferIndex + 1, OLED_FONT_MEDIUM_WIDTH), 30, OLED_FONT_MEDIUM, bufferIndex - 1, OLED_WHITE);
}
